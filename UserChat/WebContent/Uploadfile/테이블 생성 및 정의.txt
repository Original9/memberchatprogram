(
  USER_CODE NUMBER NOT NULL 
, USER_ID VARCHAR2(20) NOT NULL 
, USER_PW VARCHAR2(20) 
, GRADE_POINT NUMBER 
, SALE_POINT NUMBER 
, USER_TYPE VARCHAR2(20) 
, CONSTRAINT USER1_PK PRIMARY KEY 
  (
    USER_CODE 
  )
  ENABLE 
);

CREATE TABLE LOCATION (
  LOCATION_ID int NOT NULL 
, USER_ABLE int  --0과 1만 넣을수있게 프로그래밍(bool타입쓰려면 pl/sql쓸수있어야함)
, LOCATION_NAME VARCHAR2(20) 
, ACCESS_COUNT NUMBER NOT NULL 
, CONSTRAINT LOCATION_PK PRIMARY KEY 
  (LOCATION_ID)
  ENABLE );


CREATE TABLE GRADE 
(
  GRADE VARCHAR2(20) NOT NULL 
, MIN_COUNT NUMBER 
, CONSTRAINT GRADE_PK PRIMARY KEY 
  (
    GRADE 
  )
  ENABLE 
);


CREATE TABLE HISTORY 
(
  HISTORY NUMBER NOT NULL 
, START_TIME VARCHAR2(20) 
, END_TIME VARCHAR2(20) 
, USER_id int 
,CONSTRAINT FK_user FOREIGN KEY(USER_id)
REFERENCES user1(USER_CODE)
, USE_LOCATION int ,
CONSTRAINT FK_location FOREIGN KEY(USE_LOCATION)
REFERENCES location(location_id)
, SYS_USE NUMBER NOT NULL  --0과 1만 넣을수있게 프로그래밍(bool타입쓰려면 pl/sql쓸수있어야함)
, CONSTRAINT HISTORY_PK PRIMARY KEY 
  (
    HISTORY 
  )
  ENABLE 
);



CREATE TABLE SUGGESTIONS 
(
  SUGGESTIONS_ID NUMBER NOT NULL 
, USER_ID NUMBER ,
CONSTRAINT FK_user_id FOREIGN KEY(USER_ID)
REFERENCES user1(USER_CODE)
, SUGGESTION_DATE DATE 
, SUGGESTION_VALUE VARCHAR2(4000) 
, success VARCHAR2(2) 
, CONSTRAINT SUGGESTIONS_PK PRIMARY KEY 
  (
    SUGGESTIONS_ID 
  )
  ENABLE
